class Main {
public static void main(String[] args) {
        int n = 5;
        int [][] DimArray = new int[n][n + 2];
        boolean isOneVal = true;
        for (int [] rowArr : DimArray) {
            for (int i = 0; i < rowArr.length; i++) {
                if (isOneVal) rowArr[i] = 1;
                else rowArr[i] = 0;
                System.out.printf("%2d", rowArr[i]);
                isOneVal = !isOneVal;
            }
            System.out.println();
        }
    }
}
